# listOrderChangeActionsByTypeStep - Medusa Core Workflows Reference

This documentation provides a reference to the `listOrderChangeActionsByTypeStep`. It belongs to the `@medusajs/medusa/core-flows` package.

This step lists order change actions filtered by action type.

:::note

This is available starting from [Medusa v2.12.0](https://github.com/medusajs/medusa/releases/tag/v2.12.0).

:::

```ts title="src/workflows/my-workflow.ts"
import { createWorkflow } from "@medusajs/framework/workflows-sdk"
import { listOrderChangeActionsByTypeStep } from "@medusajs/medusa/core-flows"

const myWorkflow = createWorkflow(
  "my-workflow",
  () => {
    const data = listOrderChangeActionsByTypeStep({
      "order_change_id": "{value}",
      "action_type": {
        "FULFILL_ITEM": "{value}",
        "DELIVER_ITEM": "{value}",
        "CANCEL_ITEM_FULFILLMENT": "{value}",
        "ITEM_ADD": "{value}",
        "ITEM_REMOVE": "{value}",
        "ITEM_UPDATE": "{value}",
        "RECEIVE_DAMAGED_RETURN_ITEM": "{value}",
        "RECEIVE_RETURN_ITEM": "{value}",
        "RETURN_ITEM": "{value}",
        "CANCEL_RETURN_ITEM": "{value}",
        "SHIPPING_ADD": "{value}",
        "SHIPPING_REMOVE": "{value}",
        "SHIPPING_UPDATE": "{value}",
        "SHIP_ITEM": "{value}",
        "WRITE_OFF_ITEM": "{value}",
        "REINSTATE_ITEM": "{value}",
        "TRANSFER_CUSTOMER": "{value}",
        "UPDATE_ORDER_PROPERTIES": "{value}",
        "CREDIT_LINE_ADD": "{value}",
        "PROMOTION_ADD": "{value}",
        "PROMOTION_REMOVE": "{value}",
        "ITEM_ADJUSTMENTS_REPLACE": "{value}"
      }
    })
  }
)
```
